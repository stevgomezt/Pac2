click==8.1.3: 
# Click es una biblioteca para crear interfaces de línea de comandos en Python. 
# Proporciona herramientas para definir argumentos, opciones y comandos, 
# así como manejar la ejecución de comandos.

Flask==2.2.5: 
# Flask es un framework web ligero para Python. Permite construir aplicaciones web de forma sencilla y rápida, proporcionando funciones básicas para el enrutamiento de URLs, manejo de solicitudes y respuestas, y renderizado de plantillas.

Flask-MySQLdb==1.0.1: 
# Flask-MySQLdb es una extensión de Flask que proporciona 
# una interfaz fácil de usar para trabajar con bases de datos MySQL en aplicaciones 
# Flask. Permite conectarse a una base de datos MySQL y ejecutar consultas de manera
# conveniente.

importlib-metadata==4.12.0: 
# importlib-metadata es una biblioteca que proporciona 
# funciones para acceder a los metadatos de los paquetes de Python instalados. 
# Permite obtener información sobre los paquetes instalados, como su versión y dependencias.

itsdangerous==2.1.2: 
# itsdangerous es una biblioteca que proporciona diversas funciones
# para la seguridad en aplicaciones web. Puede utilizarse para generar y verificar
# tokens de seguridad, firmar y verificar datos, y realizar operaciones criptográficas
# seguras.

Jinja2==3.1.2: 
# Jinja2 es un motor de plantillas utilizado en el framework Flask. 
# Permite generar dinámicamente contenido HTML, XML u otros formatos de texto basado 
# en plantillas predefinidas. Es flexible y cuenta con características avanzadas 
# como herencia de plantillas y macros.

MarkupSafe==2.1.1: 
# MarkupSafe es una biblioteca que proporciona utilidades seguras 
# para procesar y escapar texto en aplicaciones web. Se utiliza para garantizar que 
# los datos ingresados por los usuarios no se interpreten como código HTML malicioso, 
# evitando así ataques de inyección.

mysqlclient==2.1.1: 
# mysqlclient es una biblioteca de Python que proporciona una 
# interfaz para conectarse y trabajar con bases de datos MySQL. 
# Permite ejecutar consultas SQL, recuperar resultados y realizar transacciones 
# con la base de datos.

PyMySQL==1.0.2: 
# PyMySQL es otra biblioteca de Python para trabajar con bases 
# de datos MySQL. Proporciona una interfaz similar a mysqlclient y permite realizar 
# operaciones como la ejecución de consultas y la gestión de conexiones con la 
# base de datos.

Werkzeug==2.2.2: 
# Werkzeug es una biblioteca utilizada en el framework Flask. 
# Proporciona herramientas para la gestión de solicitudes y respuestas HTTP, 
# manipulación de URL, enrutamiento y otras funciones relacionadas con el desarrollo web.

zipp==3.8.0: 
# zipp es una biblioteca que proporciona funciones para trabajar con 
# archivos zip en Python. Permite leer y escribir archivos comprimidos en formato zip, 
# así como acceder a su contenido de manera eficiente.

Flask-FontAwesome==0.1.5: 
# Flask-FontAwesome es una extensión de Flask que facilita 
# la integración de iconos de FontAwesome en aplicaciones web construidas con Flask. 
# Proporciona métodos simples para incluir iconos en las plantillas de Flask.
